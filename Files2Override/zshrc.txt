# Enable Powerlevel10k instant prompt. Should stay close to the top of ~/.zshrc.
# Initialization code that may require console input (password prompts, [y/n]
# confirmations, etc.) must go above this block; everything else may go below.
if [[ -r "${XDG_CACHE_HOME:-$HOME/.cache}/p10k-instant-prompt-${(%):-%n}.zsh" ]]; then
  source "${XDG_CACHE_HOME:-$HOME/.cache}/p10k-instant-prompt-${(%):-%n}.zsh"
fi

# Disable sessions file
setopt sharehistory

# Set history size
HISTSIZE=8
SAVEHIST=0

# Set terminal style
# PROMPT='%B%F{magenta}%n%f%b %1~ %# '

# Configure pyenv
eval "$(pyenv init -)"

# Enable colorful output for ls command
export CLICOLOR=1

# Define some useful functions
## Open System Prefences
systemPrefs()
{
    open -b com.apple.systempreferences
}

## Open Firefox
openFirefox()
{
   open -b org.mozilla.firefox
}

## Open Spotify
openSpotify() 
{
    open -b com.spotify.client
}

## Open LuLu
firewall()
{
    open -b com.objective-see.lulu.app
}

## Wifi toggle
wifiToggle() 
{
    networksetup -setairportpower en0 $1
}

## List processes that are listening to a port
whoIsListening()
{
    sudo lsof -i -P | grep LISTEN
}

## Show filesystem activity
fsActivity()
{
    sudo fs_usage
}

## Shutdown
takeNap()
{
    sudo halt
}

source ~/.powerlevel10k/powerlevel10k.zsh-theme

# To customize prompt, run `p10k configure` or edit ~/.p10k.zsh.
[[ ! -f ~/.p10k.zsh ]] || source ~/.p10k.zsh
